import { action } from '@storybook/addon-actions';

<Meta
  title="Task"
  component={Task}
  excludeStories={/.*Data$/}
/>

  onPinTask: action('pin-task'),
  onArchiveTask: action('archive-task'),
};

  components: { Task },
  props: Object.keys(argTypes),
  methods: actionsData,
  template: '<Task v-bind="$props" @pin-task="onPinTask" @archive-task="onArchiveTask" />',
});
